RoamEaseProject/
│
├── lib/
│   └── sqlite-jdbc-3.45.1.0.jar      # Required for database connection
│
├── database/
│   └── roamease_database.db          # Auto-generated by SQLite
│
└── src/
    └── com/cust/roamease/
        │
        ├── Main.java                 # The entry point that boots the application
        │
        ├── models/                   # Encapsulated Data Objects (OOP)
        │   ├── User.java             # (Fields: userId, name, email)
        │   ├── Trip.java             # (Fields: tripId, destination, startDate, endDate)
        │   ├── ItineraryEvent.java   # (Fields: eventId, tripId, title, datetime, location)
        │   └── Expense.java          # (Fields: expenseId, tripId, payerId, amount, description)
        │
        ├── interfaces/               # SOLID Principle Implementation
        │   ├── ICalculable.java      # (Method: calculateSplit())
        │   └── IPersistable.java     # (Method: save())
        │
        ├── db/                       # Data Access Layer (SQL Queries)
        │   ├── DatabaseManager.java  # (SQLite connection logic)
        │   ├── UserDAO.java          # (Insert/Select Users)
        │   ├── TripDAO.java          # (Manage trips and itineraries)
        │   └── ExpenseDAO.java       # (Log financial data)
        │
        ├── services/                 # Business Logic (The brain of the app)
        │   ├── TripService.java      # (Logic: creating trips, adding waypoints)
        │   └── ExpenseManager.java   # (Logic: implements ICalculable for bill splitting)
        │
        └── ui/                       # Presentation Layer
            └── ConsoleApp.java       # (Scanner menus: "Press 1 to Add Trip", etc.)
